Ternary conditionals

Don't use this:

var isArthur = false;
var weapon;
if(isArthur){
  weapon = Excalibur;
} else {
  weapon = Longsword;
}

Use this instead:

somecondition ? pickThisIfTrue : pickThisIfFalse;

var isArthur = false;
var weapon;
isArthur ? weapon = "Excalibur" : weapon = "Longsword";
